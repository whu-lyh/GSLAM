585 inline const char* Svar::MatchingEndBrace(const char* str, char flag) {
586  char endflag = '}';
587  if (flag == '(')
588  endflag = ')';

605 inline std::vector<std::string> Svar::ChopAndUnquoteString(std::string s) {
606  using namespace std;
607  vector<string> v;
608  string::size_type nPos = 0;
609  string::size_type nLength = s.length();
610  while (1) {
611  string sTarget;
612  char cDelim;
613  // Get rid of leading whitespace:
614  while ((nPos < nLength) && (s[nPos] == ' ')) nPos++;
615  if (nPos == nLength) return v;
616 
617  // First non-whitespace char...
618  if (s[nPos] != '\"')
619  cDelim = ' ';
620  else {
621  cDelim = '\"';
622  nPos++;
623  }
624  for (; nPos < nLength; ++nPos) {
625  char c = s[nPos];
626  if (c == cDelim) break;
627  if (cDelim == '"' && nPos + 1 < nLength && c == '\\') {
628  char escaped = s[++nPos];
629  switch (escaped) {
630  case 'n':
631  c = '\n';
632  break;
633  case 'r':
634  c = '\r';
635  break;
636  case 't':
637  c = '\t';
638  break;
639  default:
640  c = escaped;
641  break;
642  }
643  }
644  sTarget += c;
645  }
646  v.push_back(sTarget);
647 
648  if (cDelim == '\"') nPos++;
649  }
650 }

766  while (true) {

769  const char* endbrace = MatchingEndBrace(brace, flag);

775  expandVal(std::string(brace + 1, endbrace), flag);

801 inline bool Svar::setvar(std::string s) {
802  using namespace std;
803  // Execution failed. Maybe its an assignment.
804  string::size_type n;
805  n = s.find("=");
806  bool shouldOverwrite = true;
807 
808  if (n != string::npos) {
809  string var = s.substr(0, n);
810  string val = s.substr(n + 1);
811 
812  // Strip whitespace from around var;
813  string::size_type s = 0, e = var.length() - 1;
814  if ('?' == var[e]) {
815  e--;
816  shouldOverwrite = false;
817  }
818  for (; isspace(var[s]) && s < var.length(); s++) {
819  }
820  if (s == var.length()) // All whitespace before the `='?
821  return false;
822  for (; isspace(var[e]); e--) {
823  }
824  if (e >= s) {
825  var = var.substr(s, e - s + 1);
826 
827  // Strip whitespace from around val;
828  s = 0, e = val.length() - 1;
829  for (; isspace(val[s]) && s < val.length(); s++) {
830  }
831  if (s < val.length()) {
832  for (; isspace(val[e]); e--) {
833  }
834  val = val.substr(s, e - s + 1);
835  } else
836  val = "";
837 
838  insert(var, val, shouldOverwrite);
839  return true;
840  }
841  }
842 
843  return false;
844 }

903 inline bool Svar::ParseLine(std::string s, bool bSilentFailure) {
904  using namespace std;
905  if (s == "") return 0;
906  int& collectFlag = GetInt("Svar.Collecting", 0);
907  if (collectFlag) {
908  istringstream ist(s);
909  string sCommand;
910  ist >> sCommand;
911  if (sCommand == "endif" || sCommand == "fi") command().Call("endif");
912  if (sCommand == "else")
913  command().Call("else");
914  else if (sCommand == "endfunction")
915  command().Call("endfunction");
916  else if (sCommand == ".") {
917  command().Call(".", ist.str());
918  } else
919  command().Call(".", s);
920  return 0;
921  }
922  s = UncommentString(expandVal(s, '{'));
923  s = UncommentString(expandVal(s, '('));
924  if (s == "") return 0;
925 
926  // Old ParseLine code follows, here no braces can be left (unless in arg.)
927  istringstream ist(s);
928 
929  string sCommand;
930  string sParams;
931 
932  // Get the first token (the command name)
933  ist >> sCommand;
934  if (sCommand == "") return 0;
935 
936  // Get everything else (the arguments)...
937 
938  // Remove any whitespace
939  ist >> ws;
940  getline(ist, sParams);
941 
942  // Attempt to execute command
943  if (command().Call(sCommand, sParams)) return true;
944 
945  if (setvar(s)) return 1;
946 
947  if (!bSilentFailure)
948  cerr << "? GUI_impl::ParseLine: Unknown command \"" << sCommand
949  << "\" or invalid assignment." << endl;
950  return 0;
951 }

963  while (getline(is, buffer) && shouldParse) {
964  // Lines ending with '\' are taken as continuing on the next line.
